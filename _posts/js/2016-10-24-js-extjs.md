##### widgetcolumn

```
xtype: 'widgetcolumn',
dataIndex: 'progressuser',
text: '操作',
tdCls: 'btngroup',
width: 150,
widget: {
  xtype: 'buttongroup',
  baseCls: '',
  layout: {
    type: "column"
  },
  items: [{
    xtype: "button",
    text: '退货入库',
    handler: 'returnAndReceiptHandler',
    iconCls: "btn-edit",
  }, {
    xtype: "button",
    text: '处理', 
    handler: 'dealWithHandler',
    iconCls: "btn-edit",
  }, {
    xtype: "button",
    handler: 'refreshHandler',
    iconCls: "btn-edit",
    text: '刷新匹配',
  }, {
    xtype: "button",
  　 text: '查看', 
  　　handler: 'checkHandler',
    iconCls: "btn-edit",
  }],

  defaultBindProperty: 'system',
  setSystem: function () {
      var buttonGroup = this,
    　　btnArray = buttonGroup.items.items,
    　　record = btnArray[0].ownerCt.getWidgetRecord();
      for (var i = 0; i < btnArray.length; i++) {
        btnArray[i].hide();
      }
      var operation = record.get('operation');
      if (operation === 'MATCHED') {
        btnArray[0].show();
        btnArray[1].show();
      } else if (operation === 'NOT_MATCHED') {
        btnArray[2].show();
        btnArray[1].show();
      } else if (operation === 'TRANSFERED_IN') {
        btnArray[3].show();
      } else{

      }
  },
},
```
